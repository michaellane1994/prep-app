Setup Guide
1
Open Google Sheets, create a new spreadsheet named "Prep App Data"
2
Go to Extensions → Apps Script, paste the script below, then deploy as a Web App (Execute as: Me, Access: Anyone)
3
Copy the deployment URL, paste above, and click Test to verify. Auto-sync pushes after every change.
// Paste into Google Apps Script (Extensions → Apps Script) const SHEETS = { ingredients: 'Ingredients', recipes: 'Recipes' }; function doPost(e) { const lock = LockService.getScriptLock(); lock.tryLock(10000); try { const data = JSON.parse(e.postData.contents); const ss = SpreadsheetApp.getActiveSpreadsheet(); if (data.action === 'push') { writeSheet(ss, SHEETS.ingredients, ['id','name','category','isStaple','dateAdded'], (data.ingredients || []).map(i => [i.id, i.name, i.category, i.isStaple, i.dateAdded])); writeSheet(ss, SHEETS.recipes, ['id','name','emoji','prepTime','cookTime','servings','ingredients', 'instructions','tags','isSaved','timesCooked','lastCooked'], (data.recipes || []).map(r => [r.id, r.name, r.emoji, r.prepTime, r.cookTime, r.servings, JSON.stringify(r.ingredients), r.instructions, (r.tags||[]).join(','), r.isSaved, r.timesCooked, r.lastCooked])); return buildResponse({ ok: true }); } if (data.action === 'pull') { const ingredients = readSheet(ss, SHEETS.ingredients, r => ({ id: r.id, name: r.name, category: r.category, isStaple: String(r.isStaple).toLowerCase() === 'true', dateAdded: r.dateAdded })); const recipes = readSheet(ss, SHEETS.recipes, r => ({ id: r.id, name: r.name, emoji: r.emoji, prepTime: parseInt(r.prepTime)||0, cookTime: parseInt(r.cookTime)||0, servings: parseInt(r.servings)||2, ingredients: JSON.parse(r.ingredients||'[]'), instructions: r.instructions, tags: (r.tags||'').split(',').filter(Boolean), isSaved: String(r.isSaved).toLowerCase() === 'true', timesCooked: parseInt(r.timesCooked)||0, lastCooked: r.lastCooked })); return buildResponse({ ok: true, ingredients, recipes }); } if (data.action === 'ping') return buildResponse({ ok: true }); return buildResponse({ ok: false, error: 'Unknown action' }); } catch(err) { return buildResponse({ ok: false, error: err.toString() }); } finally { lock.releaseLock(); } } function writeSheet(ss, name, headers, rows) { let s = ss.getSheetByName(name) || ss.insertSheet(name); s.clearContents(); const d = rows.length ? [headers,...rows] : [headers]; s.getRange(1,1,d.length,headers.length).setValues(d); } function readSheet(ss, name, mapper) { const s = ss.getSheetByName(name); if (!s) return []; const rows = s.getDataRange().getValues(); if (rows.length < 2) return []; const h = rows[0]; return rows.slice(1).map(row => { const r={}; h.forEach((k,i)=>r[k]=row[i]); return mapper(r); }); } function buildResponse(obj) { return ContentService.createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); }
